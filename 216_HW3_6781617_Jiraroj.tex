\section*{Problem 1 (11.3 Dot product)}
Use dot product to find the angle that vector $v = -\sqrt{3} i + j$ makes with the vector $w = i$, (positive $x$-axis).

\medskip
\solution

we use the relation of dot product and the angles as
\[
    \cos \theta = \frac{v \cdot w}{\lVert v \rVert \lVert w \rVert}
\]

\[
    \cos \theta = \frac{-\sqrt{3} + 0}{2 \cdot 1} = \frac{-\sqrt{3}}{2}
\]
chucking basic angles --- in the second quadrant for negative cosine value, we'll get \boxed{\theta = \frac{5\pi}{6}}.

\section*{Problem 2 (11.3 Dot product)}
\begin{enumerate}[label=(\alph*)]
    \item Show that $(2,1,6)$, $(4,7,9)$ and $(11,7,-12)$ are the vertices of a right triangle.
    \item Find the area of the triangle.
\end{enumerate}

\medskip
\solution

\begin{enumerate}[label=(\alph*)]
      \item if we plot those three points in 3 dimensional space (on physical paper as well) ---
  call them $P_1$, $P_2$ and $P_3$ accordingly, we'll see that point $(2, 1, 6)$ and $(4, 7, 9)$
  are probable of being the right angle vertex, so we'll test these two.

      \begin{center}
          \begin{tikzpicture}
              \begin{axis}[
                  width=6cm,
                  height=6cm,
                  view={60}{30},
                  axis lines=center,
                  axis line style={line width=0.9pt},
                  axis on top,
                  ticks=none,
                  xlabel={$x$}, ylabel={$y$}, zlabel={$z$},
                  xmin=0, xmax=12,
                  ymin=0, ymax=12,
                  zmin=-24, zmax=12,
              ]
                  \addplot3[only marks, mark=*, mark size=3pt] coordinates {(2,1,6)};
                  \addplot3[only marks, mark=*, mark size=3pt] coordinates {(4,7,9)};
                  \addplot3[only marks, mark=*, mark size=3pt] coordinates {(11,7,-12)};
                  \addplot3[thick] coordinates {(2,1,6) (4,7,9)};
                  \addplot3[thick] coordinates {(4,7,9) (11,7,-12)};
                  \addplot3[thick] coordinates {(11,7,-12) (2,1,6)};
                  \node[anchor=south west] at (axis cs:2,1,6) {$P_1$};
                  \node[anchor=south east] at (axis cs:4,7,9) {$P_2$};
                  \node[anchor=north] at (axis cs:11,7,-12) {$P_3$};
              \end{axis}
          \end{tikzpicture}
      \end{center}

      we test $P_1$ first, we can derive the two vectors from $P_1$ to $P_2$ and $P_3$ as

      \[
          \overrightarrow{P_1P_2} = (4, 7, 9) - (2, 1, 6) = \langle 2, 6, 3 \rangle
      \]
      \[
          \overrightarrow{P_1P_3} = (11, 7, -12) - (2, 1, 6) = \langle 9, 6, -18 \rangle
      \]

      to check for right angle (orthogonal vectors), we use the dot product relation to angle as

      \begin{align*}
          \cos \frac{\pi}{2} &= \overrightarrow{P_1P_2} \cdot \overrightarrow{P_1P_3} \\
          &= \langle 2, 6, 3 \rangle \cdot \langle 9, 6, -18 \rangle \\
          &= 2 \cdot 9 + 6 \cdot 6 + 3 \cdot (-18) \\
          &= 18 + 36 - 54 \\
          &= 0
      \end{align*}

      the dot product matches the expected result when the two vector are perpendicular, so we can conclude that these three points are the vertices of a right triangle. 
      \qed

    \item since these three points formed a right triangle, finding the area is quite simple
    \[
        \begin{aligned}
            \text{Area} &= \frac{1}{2} \cdot \lVert \overrightarrow{P_1P_2} \rVert \cdot \lVert \overrightarrow{P_1P_3} \rVert \\
            &= \frac{1}{2} \cdot \sqrt{(2)^2 + (6)^2 + (3)^2} \cdot \sqrt{(9)^2 + (6)^2 + (-18)^2} \\
            &= \frac{1}{2} \cdot 7 \cdot 21 \\
            &= \boxed{\frac{147}{2} \text{ unit}^3}
        \end{aligned}
    \]
\end{enumerate}

\section*{Problem 3 (11.3 Dot product)}
Given the points $P_1 = (2, 1, 6)$, $P_2 = (4, 7, 9)$, and $P_3 = (11, 7, -12)$.
\begin{enumerate}[label=(\alph*)]
    \item Find the vector with length $2$ and has opposite direction to the vector $\overrightarrow{P_1P_2}$.
    \item Show that the vector $\overrightarrow{P_1P_2}$ is orthogonal (perpendicular) to the vector $\overrightarrow{P_1P_3}$.
\end{enumerate}

\medskip
\solution

\begin{enumerate}[label= (\alph*)]
    \item we first have to find the unit vector of $\overrightarrow{P_1P_2}$ as
    \[
        \begin{aligned}
            \widehat{\overrightarrow{P_1P_2}} &= \frac{\overrightarrow{P_1P_2}}{\lVert \overrightarrow{P_1P_2} \rVert} \\
            &= \frac{\langle 4 - 2, 7 - 1, 9 - 6 \rangle}{\sqrt{(4 - 2)^2 + (7 - 1)^2 + (9 - 6)^2}} \\
            &= \frac{\langle 2, 6, 3 \rangle}{\sqrt{(2)^2 + (6)^2 + (3)^2}} \\
            &= \frac{1}{7} \langle 2, 6, 3 \rangle.
        \end{aligned}
    \]


    a unit vector is well ... a vector with magnitude of 1; so, a vector with length 2, which has opposite direction to $\overrightarrow{P_1P_2}$ can be created by multiply $-2$ to $\widehat{\overrightarrow{P_1P_2}}$ as
    \[
        \boxed{-\frac{2}{7} \langle 2, 6, 3 \rangle = \left\langle -\frac{4}{7}, -\frac{12}{7}, -\frac{6}{7} \right\rangle}
    \]

    \newpage
    \item we already find $\overrightarrow{P_1P_2}$, so we only have to find $\overrightarrow{P_1P_3}$ as
    \[
        \overrightarrow{P_1P_3} = (11, 7, -12) - (2, 1, 6) = \langle 9, 6, -18 \rangle
    \]

    a perpendicular vectors must have a dot product of 0, which 

    \[
        \begin{aligned}
            \overrightarrow{P_1P_2} \cdot \overrightarrow{P_1P_3} &= \langle 2, 6, 3 \rangle \cdot \langle 9, 6, -18 \rangle \\
            &= 2 \cdot 9 + 6 \cdot 6 + 3 \cdot (-18) \\
            &= 18 + 36 - 54 \\
            &= 0
        \end{aligned}
    \]

    so, we can conclude that $\overrightarrow{P_1P_2} \text{ and } \overrightarrow{P_1P_3}$ are perpendicular. 
    \qed
\end{enumerate}

\section*{Problem 4 (11.3 Dot product)}
True-False Determine whether the statement is true or false. Explain your answer.
\begin{enumerate}[label=(\alph*)]
    \item If $v \cdot u = v \cdot w$ and $v \neq 0$, then $u = w$.
    \item If $u$ is a unit vector that is parallel to a nonzero vector $v$, then $u \cdot v = \pm \lVert v \rVert$.
\end{enumerate}

\medskip
\solution

\begin{enumerate}[label=(\alph*)]
    \item let $v = \langle 1, 1, 1, \dots \rangle$, the vector whose coordinates are all $1$'s in n dimensions. Pick $n$ scalars $\{a_1, a_2, \dots, a_n\}$ and form two vectors $u = \langle a_1, a_2, \dots, a_n \rangle$ and $w = \langle a_{\sigma(1)}, a_{\sigma(2)}, \dots, a_{\sigma(n)} \rangle$ where $\sigma$ is any nontrivial permutation of $\{1, \dots, n\}$. Then $v \cdot u = v \cdot w = \sum_{i = 0}^{n} a_i$, yet $u \neq w$, so the statement is \boxed{false}.
    \item from the geometric definition of dot product: $u \cdot v = \lVert u \lVert \lVert v \lVert \cos\theta$ --- which can be interpret as the scalar projection of u on v, times the magnitude of v. the scalar projection of u on v can be written as 
        \[u_v = \lVert u \lVert \cos\theta = u \cdot \hat{v}\]
        \[v_u = \lVert v \lVert \cos\theta = v \cdot \hat{u}\]
    
    thus 
        \[u \cdot v = u_v \cdot \lVert v \lVert\]

    for $u_v$ --- a scalar projection of a unit vector $u$ onto a parallell unit vector of $v$ will gives -1 or 1, thus if u is a unit vector that's parallel to a non-zero vector v, then $u \cdot v = \pm \lVert v \lVert$
    \qed
\end{enumerate}

\section*{Problem 5 (11.4 Cross product)}
Use determinant to find the cross product
\[
    i \times (i + j + k).
\]

\medskip
\solution

write the cross product in matrix form as
\[
    \begin{pmatrix}
        \mathbf{i} & \mathbf{j} & \mathbf{k} \\
        1 & 0 & 0 \\
        1 & 1 & 1
    \end{pmatrix}
\]

then find det of it with chio's method of condensation

\[
    \frac{1}{\mathbf{i}}
    \begin{vmatrix}
        \mathbf{j} & \mathbf{k} \\
        \mathbf{i} - \mathbf{j} & \mathbf{i} - \mathbf{k}
    \end{vmatrix}
\]

\[
    =
    \frac{1}{\mathbf{i}}
    \left(
        \mathbf{j}(\mathbf{i} - \mathbf{k}) - \mathbf{k}(\mathbf{i} - \mathbf{j})
    \right)
    =
    \frac{1}{\mathbf{i}}
    \left(
        \mathbf{j}\mathbf{i} - \mathbf{j}\mathbf{k} - \mathbf{k}\mathbf{i} + \mathbf{k}\mathbf{j}
    \right).
\]

or using the cofactor expansion method

\[
  \mathbf{i} \cdot \det\begin{pmatrix}0&0\\1&1\end{pmatrix}
  -\mathbf{j} \cdot \det\begin{pmatrix}1&0\\1&1\end{pmatrix}
  +\mathbf{k} \cdot \det\begin{pmatrix}1&0\\1&1\end{pmatrix}
  =0\mathbf{i} -\mathbf{j} + \mathbf{k}.
\]

we'll get the cross product as \boxed{\mathbf{k} - \mathbf{j}}

\section*{Problem 6 (11.4 Cross product)}
Let $u = \langle 1, 2, -3 \rangle$, $v = \langle -4, 1, 3 \rangle$. Find $u \times v$ and check that it is orthogonal to both $u$ and $v$.

\medskip
\solution

write the cross product in matrix form as
  \[
      \begin{vmatrix}
          \mathbf{i} & \mathbf{j} & \mathbf{k} \\
          1 & 2 & -3 \\
          -4 & 1 & 3
      \end{vmatrix}
  \]

  then find det of it with chio's method

  \[
      \mathbf{i}
      \begin{vmatrix}
          2 & -3 \\
          1 & 3
      \end{vmatrix}
      - \mathbf{j}
      \begin{vmatrix}
          1 & -3 \\
          -4 & 3
      \end{vmatrix}
      + \mathbf{k}
      \begin{vmatrix}
          1 & 2 \\
          -4 & 1
      \end{vmatrix}
  \]

  \[
      =
      \mathbf{i}(2 \cdot 3 - (-3) \cdot 1)
      - \mathbf{j}(1 \cdot 3 - (-3)(-4))
      + \mathbf{k}(1 \cdot 1 - 2(-4))
      =
      9\mathbf{i} + 9\mathbf{j} + 9\mathbf{k}
  \]

  we'll get the cross product as 9\mathbf{i} + 9\mathbf{j} + 9\mathbf{k}; then to check orthogonality of the cross product to $u$ and $v$ --- we use dot product's definition of $a \cdot b = \lVert a \lVert \lVert b \lVert cos \theta \text{ ; } \theta = \pm\frac{\pi}{2}$
  \[
      (u \times v) \cdot u = \langle 9, 9, 9 \rangle \cdot \langle 1, 2, -3 \rangle = 9 + 18 - 27
  = 0
  \]
  \[
      (u \times v) \cdot v = \langle 9, 9, 9 \rangle \cdot \langle -4, 1, 3 \rangle = -36 + 9 + 27
  = 0
  \]
  so $u \times v$ is orthogonal to both $u$ and $v$ --- which is expected by cross product's definition.
